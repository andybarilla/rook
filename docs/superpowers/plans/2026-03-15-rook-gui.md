# Rook GUI Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Wails v2 desktop app with React frontend that provides a graphical interface to Rook's core library — workspace management, service orchestration, log viewing, and auto-discovery.

**Architecture:** A Go service layer (`internal/api/`) wraps the existing core packages and exposes methods via Wails bindings. The React frontend (TypeScript, Tailwind v4, shadcn/ui) calls these bindings and subscribes to real-time events for status changes and log streaming. The service layer is designed as a clean boundary for future daemon extraction.

**Tech Stack:** Go 1.22+, Wails v2, React 19, TypeScript, Tailwind CSS v4, shadcn/ui

**Spec:** `docs/specs/2026-03-15-rook-gui-design.md`

// StartService starts a single service by name.
func (o *Orchestrator) StartService(ctx context.Context, ws workspace.Workspace, serviceName string) error {
	svc, ok := ws.Services[serviceName]
	if !ok {
		return fmt.Errorf("unknown service: %q", serviceName)
	}

	o.mu.Lock()
	if o.handles[ws.Name] == nil {
		o.handles[ws.Name] = make(map[string]runner.RunHandle)
	}
	if _, running := o.handles[ws.Name][serviceName]; running {
		o.mu.Unlock()
		return nil // already running
	}
	o.mu.Unlock()

	// Build workspace-wide portMap so env templates (e.g. {{.Port.postgres}}) resolve correctly
	portMap := make(runner.PortMap)
	if o.portAllocator != nil {
		// Ensure this service has a port allocated
		if len(svc.Ports) > 0 {
			if svc.PinPort > 0 {
				port, err := o.portAllocator.AllocatePinned(ws.Name, serviceName, svc.PinPort)
				if err != nil {
					return fmt.Errorf("pinning port for %s: %w", serviceName, err)
				}
				portMap[serviceName] = port
			} else {
				port, err := o.portAllocator.Allocate(ws.Name, serviceName, svc.Ports[0])
				if err != nil {
					return fmt.Errorf("allocating port for %s: %w", serviceName, err)
				}
				portMap[serviceName] = port
			}
		}
		// Include all other services' allocated ports for template resolution
		for name := range ws.Services {
			if name == serviceName {
				continue
			}
			if result := o.portAllocator.Get(ws.Name, name); result.OK {
				portMap[name] = result.Port
			}
		}
	}

	var r runner.Runner
	if svc.IsContainer() {
		r = o.containerRunner
	} else {
		r = o.processRunner
	}

	handle, err := r.Start(ctx, serviceName, svc, portMap, ws.Root)
	if err != nil {
		return fmt.Errorf("starting %s: %w", serviceName, err)
	}

	o.mu.Lock()
	o.handles[ws.Name][serviceName] = handle
	o.mu.Unlock()

	return nil
}

// StopService stops a single service by name.
func (o *Orchestrator) StopService(ctx context.Context, ws workspace.Workspace, serviceName string) error {
	o.mu.Lock()
	handles, ok := o.handles[ws.Name]
	if !ok {
		o.mu.Unlock()
		return nil
	}
	handle, running := handles[serviceName]
	o.mu.Unlock()

	if !running {
		return nil
	}

	var r runner.Runner
	if handle.Type == "process" {
		r = o.processRunner
	} else {
		r = o.containerRunner
	}

	if err := r.Stop(handle); err != nil {
		return fmt.Errorf("stopping %s: %w", serviceName, err)
	}

	o.mu.Lock()
	delete(o.handles[ws.Name], serviceName)
	o.mu.Unlock()

	return nil
}

// RestartService stops and starts a single service.
func (o *Orchestrator) RestartService(ctx context.Context, ws workspace.Workspace, serviceName string) error {
	if err := o.StopService(ctx, ws, serviceName); err != nil {
		return err
	}
	return o.StartService(ctx, ws, serviceName)
}

// internal/api/logbuffer.go
package api

import (
	"sync"
	"time"
)

// LogLine represents a single log entry.
type LogLine struct {
	Workspace string    `json:"workspace"`
	Service   string    `json:"service"`
	Line      string    `json:"line"`
	Timestamp time.Time `json:"timestamp"`
}

// LogBuffer is a per-workspace rolling log buffer.
type LogBuffer struct {
	mu       sync.Mutex
	maxLines int
	lines    map[string][]LogLine // workspace -> lines
}

// NewLogBuffer creates a log buffer with the given per-workspace capacity.
func NewLogBuffer(maxLines int) *LogBuffer {
	return &LogBuffer{
		maxLines: maxLines,
		lines:    make(map[string][]LogLine),
	}
}

// Add appends a log line for a workspace/service.
func (b *LogBuffer) Add(workspace, service, line string) LogLine {
	entry := LogLine{
		Workspace: workspace,
		Service:   service,
		Line:      line,
		Timestamp: time.Now(),
	}

	b.mu.Lock()
	defer b.mu.Unlock()

	b.lines[workspace] = append(b.lines[workspace], entry)
	if len(b.lines[workspace]) > b.maxLines {
		b.lines[workspace] = b.lines[workspace][len(b.lines[workspace])-b.maxLines:]
	}

	return entry
}

// Get retrieves recent log lines for a workspace, optionally filtered by service.
// Pass empty string for service to get all. Returns at most `limit` lines.
func (b *LogBuffer) Get(workspace, service string, limit int) []LogLine {
	b.mu.Lock()
	defer b.mu.Unlock()

	all := b.lines[workspace]
	if service == "" {
		if len(all) <= limit {
			out := make([]LogLine, len(all))
			copy(out, all)
			return out
		}
		out := make([]LogLine, limit)
		copy(out, all[len(all)-limit:])
		return out
	}

	// Filter by service, take last `limit`
	var filtered []LogLine
	for _, l := range all {
		if l.Service == service {
			filtered = append(filtered, l)
		}
	}

	if len(filtered) <= limit {
		return filtered
	}
	return filtered[len(filtered)-limit:]
}
```

// internal/api/workspace.go
package api

import (
	"context"
	"fmt"
	"path/filepath"

	"github.com/andybarilla/rook/internal/discovery"
	"github.com/andybarilla/rook/internal/envgen"
	"github.com/andybarilla/rook/internal/orchestrator"
	"github.com/andybarilla/rook/internal/ports"
	"github.com/andybarilla/rook/internal/registry"
	"github.com/andybarilla/rook/internal/workspace"
	"gopkg.in/yaml.v3"
)

// WorkspaceAPI is the service layer bound to Wails.
type WorkspaceAPI struct {
	registry    *registry.FileRegistry
	portAlloc   *ports.FileAllocator
	orch        *orchestrator.Orchestrator
	discoverers []discovery.Discoverer
	logBuffer   *LogBuffer
	emitter     EventEmitter
	activeProfiles map[string]string // workspace name -> active profile
}

// NewWorkspaceAPI creates a new WorkspaceAPI.
func NewWorkspaceAPI(
	reg *registry.FileRegistry,
	alloc *ports.FileAllocator,
	orch *orchestrator.Orchestrator,
	discoverers []discovery.Discoverer,
) *WorkspaceAPI {
	return &WorkspaceAPI{
		registry:       reg,
		portAlloc:      alloc,
		orch:           orch,
		discoverers:    discoverers,
		logBuffer:      NewLogBuffer(10000),
		emitter:        &NoopEmitter{},
		activeProfiles: make(map[string]string),
	}
}

// SetEmitter sets the event emitter (called after Wails runtime is ready).
func (a *WorkspaceAPI) SetEmitter(e EventEmitter) {
	a.emitter = e
}

// BufferLog adds a log line and emits a log event. Public for use by runner hooks.
func (a *WorkspaceAPI) BufferLog(workspace, service, line string) {
	entry := a.logBuffer.Add(workspace, service, line)
	a.emitter.Emit("service:log", LogEvent{
		Workspace: workspace,
		Service:   service,
		Line:      line,
		Timestamp: entry.Timestamp.UnixMilli(),
	})
}

// --- Workspace management ---

func (a *WorkspaceAPI) ListWorkspaces() []WorkspaceInfo {
	entries := a.registry.List()
	result := make([]WorkspaceInfo, 0, len(entries))

	for _, e := range entries {
		info := WorkspaceInfo{
			Name: e.Name,
			Path: e.Path,
			ActiveProfile: a.activeProfiles[e.Name],
		}

		m, err := workspace.ParseManifest(filepath.Join(e.Path, "rook.yaml"))
		if err == nil {
			ws, err := m.ToWorkspace(e.Path)
			if err == nil {
				info.ServiceCount = len(ws.Services)
				statuses, _ := a.orch.Status(*ws)
				for _, s := range statuses {
					if s == "running" {
						info.RunningCount++
					}
				}
			}
		}

		result = append(result, info)
	}
	return result
}

func (a *WorkspaceAPI) GetWorkspace(name string) (*WorkspaceDetail, error) {
	entry, err := a.registry.Get(name)
	if err != nil {
		return nil, err
	}

	m, err := workspace.ParseManifest(filepath.Join(entry.Path, "rook.yaml"))
	if err != nil {
		return nil, err
	}

	ws, err := m.ToWorkspace(entry.Path)
	if err != nil {
		return nil, err
	}

	statuses, _ := a.orch.Status(*ws)

	// Build ordered service list
	order, _ := orchestrator.TopoSort(ws.Services, ws.ServiceNames())
	services := make([]ServiceInfo, 0, len(order))
	for _, name := range order {
		svc := ws.Services[name]
		si := ServiceInfo{
			Name:      name,
			Image:     svc.Image,
			Command:   svc.Command,
			Status:    statuses[name],
			DependsOn: svc.DependsOn,
		}
		if result := a.portAlloc.Get(ws.Name, name); result.OK {
			si.Port = result.Port
		}
		services = append(services, si)
	}

	// Collect profile names
	profiles := make([]string, 0, len(ws.Profiles))
	for p := range ws.Profiles {
		profiles = append(profiles, p)
	}

	return &WorkspaceDetail{
		Name:          ws.Name,
		Path:          entry.Path,
		Services:      services,
		Profiles:      profiles,
		Groups:        ws.Groups,
		ActiveProfile: a.activeProfiles[ws.Name],
	}, nil
}

func (a *WorkspaceAPI) AddWorkspace(path string) (*DiscoverResult, error) {
	result, err := discovery.RunAll(path, a.discoverers)
	if err != nil {
		return nil, err
	}

	// Register workspace
	wsName := filepath.Base(path)
	manifestPath := filepath.Join(path, "rook.yaml")

	if len(result.Services) > 0 {
		m := &workspace.Manifest{
			Name:     wsName,
			Type:     workspace.TypeSingle,
			Services: result.Services,
			Groups:   result.Groups,
		}
		if err := workspace.WriteManifest(manifestPath, m); err != nil {
			return nil, err
		}
	}

	// Parse the manifest (newly created or existing)
	m, err := workspace.ParseManifest(manifestPath)
	if err != nil {
		return nil, err
	}

	if err := a.registry.Register(m.Name, path); err != nil {
		return nil, err
	}

	// Allocate ports (one port per service, using first preferred port)
	for name, svc := range m.Services {
		if svc.PinPort > 0 {
			a.portAlloc.AllocatePinned(m.Name, name, svc.PinPort)
		} else if len(svc.Ports) > 0 {
			a.portAlloc.Allocate(m.Name, name, svc.Ports[0])
		}
	}

	a.emitter.Emit("workspace:changed", WorkspaceChangedEvent{
		Workspace: m.Name, Action: "added",
	})

	return &DiscoverResult{
		Source:   result.Source,
		Services: result.Services,
		Groups:   result.Groups,
	}, nil
}

func (a *WorkspaceAPI) RemoveWorkspace(name string) error {
	entry, err := a.registry.Get(name)
	if err != nil {
		return err
	}

	// Stop running services first
	m, err := workspace.ParseManifest(filepath.Join(entry.Path, "rook.yaml"))
	if err == nil {
		ws, err := m.ToWorkspace(entry.Path)
		if err == nil {
			a.orch.Down(context.Background(), *ws)
		}
	}

	a.registry.Remove(name)
	a.emitter.Emit("workspace:changed", WorkspaceChangedEvent{
		Workspace: name, Action: "removed",
	})
	return nil
}

		return err
	}
	a.emitter.Emit("workspace:changed", WorkspaceChangedEvent{
		Workspace: name, Action: "modified",
	})
	return nil
}

func (a *WorkspaceAPI) GetEnv(name string) (map[string][]EnvVar, error) {
	entry, err := a.registry.Get(name)
	if err != nil {
		return nil, err
	}
	m, err := workspace.ParseManifest(filepath.Join(entry.Path, "rook.yaml"))
	if err != nil {
		return nil, err
	}

	// Build port map from allocator
	portMap := make(map[string]int)
	for svcName := range m.Services {
		if result := a.portAlloc.Get(m.Name, svcName); result.OK {
			portMap[svcName] = result.Port
		}
	}

	result := make(map[string][]EnvVar)
	for svcName, svc := range m.Services {
		if len(svc.Environment) == 0 {
			continue
		}
		resolved, err := envgen.ResolveTemplates(svc.Environment, portMap, false)
		if err != nil {
			return nil, fmt.Errorf("resolving env for %s: %w", svcName, err)
		}
		var vars []EnvVar
		for k, tmpl := range svc.Environment {
			vars = append(vars, EnvVar{Key: k, Template: tmpl, Resolved: resolved[k]})
		}
		result[svcName] = vars
	}
	return result, nil
}

export function useWorkspaces() {
  const [workspaces, setWorkspaces] = useState<WorkspaceInfo[]>([])
  const [loading, setLoading] = useState(true)

  const refresh = useCallback(async () => {
    try {
      const list = await window.go.api.WorkspaceAPI.ListWorkspaces()
      setWorkspaces(list)
    } catch (e) {
      console.error('Failed to list workspaces:', e)
    } finally {
      setLoading(false)

      refresh()

// frontend/src/components/LogViewer.tsx
import { useEffect, useRef, useState, useCallback } from 'react'
import type { LogLine, LogEvent } from '../hooks/useWails'

interface LogViewerProps {
  workspaceName: string
  services: string[]
}

const SERVICE_COLORS = [
  'text-rook-running',    // green
  'text-rook-partial',    // amber
  'text-blue-400',        // blue
  'text-purple-400',      // purple
  'text-cyan-400',        // cyan
  'text-pink-400',        // pink
]

export function LogViewer({ workspaceName, services }: LogViewerProps) {
  const [activeTab, setActiveTab] = useState<string>('')
  const [lines, setLines] = useState<LogLine[]>([])
  const [filter, setFilter] = useState('')
  const [autoScroll, setAutoScroll] = useState(true)
  const scrollRef = useRef<HTMLDivElement>(null)

  const colorMap = useCallback((service: string) => {
    const idx = services.indexOf(service) % SERVICE_COLORS.length
    return SERVICE_COLORS[idx >= 0 ? idx : 0]
  }, [services])

  // Fetch initial logs
  useEffect(() => {
    const service = activeTab || ''
    window.go.api.WorkspaceAPI.GetLogs(workspaceName, service, 500)
      .then(setLines)
      .catch(console.error)
  }, [workspaceName, activeTab])

  // Subscribe to live logs
  useEffect(() => {
    const off = window.runtime.EventsOn('service:log', (event: LogEvent) => {
      if (event.workspace !== workspaceName) return
      if (activeTab && event.service !== activeTab) return

      const line: LogLine = {
        workspace: event.workspace,
        service: event.service,
        line: event.line,
        timestamp: event.timestamp,
      }
      setLines(prev => [...prev.slice(-999), line])
    })
    return off
  }, [workspaceName, activeTab])

  // Auto-scroll
  useEffect(() => {
    if (autoScroll && scrollRef.current) {
      scrollRef.current.scrollTop = scrollRef.current.scrollHeight
    }
  }, [lines, autoScroll])

  const handleScroll = () => {
    if (!scrollRef.current) return
    const { scrollTop, scrollHeight, clientHeight } = scrollRef.current
    setAutoScroll(scrollHeight - scrollTop - clientHeight < 50)
  }

  const filteredLines = filter
    ? lines.filter(l => l.line.toLowerCase().includes(filter.toLowerCase()))
    : lines

  return (
    <div className="flex flex-col h-full">
      {/* Tab bar */}
      <div className="flex border-b border-rook-border px-4">
        <TabButton
          label="All"
          active={activeTab === ''}
          onClick={() => setActiveTab('')}
        />
        {services.map(s => (
          <TabButton
            key={s}
            label={s}
            active={activeTab === s}
            onClick={() => setActiveTab(s)}
            className={colorMap(s)}
          />
        ))}
      </div>

      {/* Filter */}
      <div className="px-4 py-1.5 border-b border-rook-border">
        <input
          type="text"
          placeholder="Filter logs..."
          value={filter}
          onChange={e => setFilter(e.target.value)}
          className="w-full bg-rook-input text-rook-text-secondary border border-rook-border rounded px-2 py-1 text-xs"
        />
      </div>

      {/* Log output */}
      <div
        ref={scrollRef}
        onScroll={handleScroll}
        className="flex-1 overflow-auto px-4 py-2 font-mono text-[9px] leading-relaxed bg-[#111122]"
      >
        {filteredLines.map((l, i) => (
          <div key={`${l.timestamp}-${l.service}-${i}`}>
            {activeTab === '' && (
              <span className={colorMap(l.service)}>
                [{l.service.padEnd(12)}]
              </span>
            )}{' '}
            <span className="text-rook-text-secondary">{l.line}</span>
          </div>
        ))}
      </div>

      {/* Jump to bottom */}
      {!autoScroll && (
        <button
          onClick={() => {
            setAutoScroll(true)
            scrollRef.current?.scrollTo({ top: scrollRef.current.scrollHeight })
          }}
          className="absolute bottom-4 right-4 bg-rook-accent text-white text-xs px-3 py-1 rounded"
        >
          Jump to bottom
        </button>
      )}
    </div>
  )
}

function TabButton({ label, active, onClick, className = '' }: {
  label: string; active: boolean; onClick: () => void; className?: string
}) {
  return (
    <button
      onClick={onClick}
      className={`px-2.5 py-1.5 text-[10px] border-b-2 ${
        active
          ? 'text-rook-text border-rook-accent font-semibold'
          : `${className || 'text-rook-muted'} border-transparent`
      }`}
    >
      {label}
    </button>
  )
}

              <p className="text-rook-running text-sm mb-3">Workspace added!</p>
              {result.source && (
                <p className="text-rook-muted text-xs mb-2">Discovered from: {result.source}</p>
              )}
              {result.services && Object.keys(result.services).length > 0 && (
                <div className="space-y-1 mb-4">
                  {Object.entries(result.services).map(([name, svc]: [string, any]) => (
                    <div key={name} className="bg-rook-card rounded px-3 py-2 text-xs">
                      <span className="text-rook-text font-semibold">{name}</span>
                      <span className="text-rook-muted ml-2">
                        {svc.image || svc.command || 'service'}
                      </span>
                    </div>
                  ))}
                </div>
              )}
              <button
                onClick={handleDone}
                className="bg-rook-accent text-white px-4 py-2 rounded text-sm w-full"
              >
                Done
              </button>
            </>
          )}

          {status === 'error' && (
            <>
              <p className="text-rook-crashed text-sm mb-3">{error}</p>
              <button
                onClick={() => { setStatus('input'); setError(null) }}
                className="text-rook-muted text-xs hover:underline"
              >
                Try again
              </button>
            </>
          )}
        </div>
      </div>
    </div>
  )
}
```
